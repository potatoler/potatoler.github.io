题目描述

n+1n+1n+1 个点从 111 到 n+1n+1n+1，每次有 pi100\frac{p_i}{100}100pi​​ 的机率前往下一个点，没有前往下一个点就回到 111 号点重新出发，求到达 n+1n+1n+1 号点的期望步数。

思路

弱化版「线形生物」，相当于每个点都只有一条向 111 的返祖边（顺便安利自己的题解）。

设 fif_ifi​ 为从 iii 到 i+1i+1i+1 的期望步数，有 f_i=\frac{p_i}{100}×1+\frac{100-p_i}{100}×(1+sum+f_i)，其中 sumsumsum 是 fif_ifi​ 前缀和。

上面的式子化简得到 fi=100+(100−pi)×sumpif_i=\frac{100+(100-p_i)×sum}{p_i}fi​=pi​100+(100−pi​)×sum​，线性递推即可。

Code

#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<climits>
#include<algorithm>
#define int long long
using namespace std;
const int MaxN = 200005, Mod = 998244353;
int n, p[MaxN], sum;

inline int Read(){
	int num = 0, op = 1;
	char ch = getchar();
	while(!isdigit(ch)){
		if(ch == '-') op = -1;
		ch = getchar();
	}
	while(isdigit(ch)){
		num = num * 10 + ch - '0';
		ch = getchar();
	}
	return num * op;
}

inline int QuickPower(int a, int b){
	int res = 1;
	while(b){
		if(b & 1) res = res * a % Mod;
		a = a * a % Mod;
		b >>= 1;
	}
	return res % Mod;
}

signed main(){
	n = Read();
	for(int i=1; i<=n; i++){
		p[i] = Read();
		p[i] = (100 + (100-p[i]) * sum % Mod) * QuickPower(p[i], Mod-2) % Mod;
		sum = (sum + p[i]) % Mod;
	}
	printf("%lld", sum);
	return 0;
}
